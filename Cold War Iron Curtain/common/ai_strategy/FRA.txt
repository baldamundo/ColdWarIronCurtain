FRA_build_peacetime_army = {
	allowed = { tag = FRA }
	enable = { 
		date > 1949.1.1
		has_war = no
		OR = { 	has_army_manpower = { size < 300000 }
				strength_ratio = { tag = SOV ratio < 0.14 } 
				strength_ratio = { tag = PRC ratio < 0.1 }
				strength_ratio = { tag = USA ratio < 0.4 }
				strength_ratio = { tag = ENG ratio < 1.2 }
				strength_ratio = { tag = WGR ratio < 1.2 }    
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = build_army id = motorized value = 500 }
	ai_strategy = { type = build_army id = infantry value = 400 }
	ai_strategy = { type = force_build_armies value = 300 }
}
