RAJ_build_peacetime_army = {
	allowed = { tag = RAJ }
	enable = { 
		date > 1949.1.1
		has_war = no
		OR = { 	has_army_manpower = { size < 2000000 }
				strength_ratio = { tag = PAK ratio < 2 }
				strength_ratio = { tag = PRC ratio < 0.66 }  
			}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = build_army id = motorized value = 100 }
	ai_strategy = { type = build_army id = infantry value = 100 }
	ai_strategy = { type = force_build_armies value = 90 }
}
